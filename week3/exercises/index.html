1. Write a Bubble Sort algorithm in JavaScript. Note: Bubble sort is a simple sorting algorithm that works by repeatedly stepping through the list to be sorted, Sample Data: [6,4,0, 3,-2,1] Expected Output : [-2, 0, 1, 3, 4, 6]

Codes

function bubbleSort(arr) {
    var sorted = false;
    while (!sorted) {
        sorted = true;
        for (var x = 0; x < arr.length - 1; x++) {
            if (arr[x] > arr[x + 1]) {
                var y = arr[x + 1];
                arr[x + 1] = arr[x];
                arr[x] = y;
                sorted = false;
            }
        }
    }
    return arr;
};

function execute() {
    let arraux = [6, 4, 0, 3, -2, 1];
    document.getElementById("result01").innerHTML = bubbleSort(arraux);
}
