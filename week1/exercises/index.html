function is_palindrome() {
    let count = 0;
    let el = document.getElementById("ex02");
    let entry = el.value;
    let str = entry.toLowerCase().replace(/[^a-zA-Z0-9]+/g, '');
    if (str === "") {
        document.getElementById('result02').value = "Nothing found!";
        return false;
    }
    let rest = (str.length) % 2;
    if (rest === 0) {
        count = (str.length) / 2;
    } else {
        if (str.length === 1) {
            document.getElementById('result02').value = "Entry is a palindrome.";
            return true;
        } else {
            count = (str.length - 1) / 2;
        }
    }
    for (var x = 0; x < count; x++) {
        if (str[x] != str.slice(-1 - x)[0]) {
            document.getElementById('result02').value = "Entry is not a palindrome.";
            return false;
        }
    }
    document.getElementById('result02').value = "The entry is a palindrome.";
    return true;
}
